        int main() {

            // Performing arithmetic operations
            int sum = a + b;           // Addition
            int difference = a - b;    // Subtraction
            int product = a * b;       // Multiplication
            int quotient = a / b;      // Division
            int remainder = a % b;     // Modulus
        
            // Displaying the results
            printf("Addition: %d + %d = %d\n", a, b, sum);
            printf("Subtraction: %d - %d = %d\n", a, b, difference);
            printf("Multiplication: %d * %d = %d\n", a, b, product);
            printf("Division: %d / %d = %d\n", a, b, quotient);
            printf("Modulus: %d %% %d = %d\n", a, b, remainder);
        
            return 0;
        }
        
  
  
  
  কেন C প্রোগ্রামিং-এ %% ব্যবহার হয় ?
  
  C প্রোগ্রামিং-এ %% ব্যবহার
  C প্রোগ্রামিং-এ, দুইটি শতাংশ চিহ্ন (%%) printf বিবৃতিতে একটি একক শতাংশ চিহ্ন (%) প্রদর্শন করার জন্য ব্যবহৃত হয়।
  এটি অবশিষ্টাংশ অপারেটরের সাথে সম্পর্কিত নয়, বরং printf ফাংশনের একটি ফরম্যাটিং বৈশিষ্ট্য।
  
  printf-এ %% ব্যবহার ব্যাখ্যা
  যখন আপনি আউটপুটে একটি শতাংশ চিহ্ন প্রদর্শন করতে চান,
  তখন আপনাকে এটি %% ব্যবহার করে স্কেপ করতে হয়। 
  এটি printf-কে জানায় যে এটি একটি অক্ষর হিসেবে বিবেচিত হবে, ফরম্যাট স্পেসিফায়ারের পরিবর্তে।

Example 


    C প্রোগ্রামিংয়ে ++A এবং A++ হল দুটি ইনক্রিমেন্ট অপারেটর।

    ++A (প্রিফিক্স ইনক্রিমেন্ট): A-এর মান 1 বাড়ায়, এবং তারপর নতুন মান ফেরত দেয়।
    A++ (পোস্টফিক্স ইনক্রিমেন্ট): বর্তমান A-এর মান ফেরত দেয়, পরে A-এর মান 1 বাড়ায়।
    
   
    #include <stdio.h>
    
      int main() {
          int A = 5;
          int B = ++A; // এখানে A-এর মান 1 বাড়বে, B হবে 6, A হবে 6
          printf("B = %d, A = %d\n", B, A);
      
          int C = A++; // এখানে C হবে 6 (কারণ A এর পুরানো মান), তারপর A হবে 7
          printf("C = %d, A = %d\n", C, A);
      
          return 0;
      }
      

      //আউটপুট:
    
      B = 6, A = 6
      C = 6, A = 7
